\begin{document}


\begin{tikzpicture}

    \begin{axis}[
        height=7cm,width=20cm,
        axis y line*=left,
        xmin=1300,xmax=1700,
        ylabel={ground-truth outputs},
        xlabel={time steps, $n$}
    ]{
        \addplot+[color=red,mark=.] table[x index={0}, y index={2}]{\mydata};
        \label{inputs};
    }
    \end{axis}
    
    \begin{axis}[
        height=7cm,width=20cm,
        axis y line*=right,
        ytick pos=right,
        ylabel near ticks,
        ylabel={predicted outputs},     
        xmin=1300,xmax=1700,
        legend pos=north west,
    ]{
        \addlegendimage{/pgfplots/refstyle=inputs}\addlegendentry{ground-truth outputs}
        \addplot+[color=blue,mark=.] table[x index={0}, y index={9}]{\mydata};
        \addlegendentry{predictive outputs}   
    }\end{axis}
