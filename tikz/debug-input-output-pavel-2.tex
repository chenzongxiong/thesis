    ]{
        \addplot+[color=red,mark=.] table[x index={0}, y index={1}]{\mydata};

    }
    \end{axis}

        legend pos=north west,
    ]{
        \addlegendimage{/pgfplots/refstyle=inputs}\addlegendentry{P-inputs}
        \addplot+[color=blue,mark=.] table[x index={0}, y index={2}]{\mydata};
